**Add spill method (demoteToSsd):**

/// Spill RAM block to SSD tier.
/// Writes block data to SSD file at fixed offset (block_id × block_bytes).
fn demoteToSsd(self: *TieredKvCache, block_id: u32) !void {
    var blk = &self.blocks[block_id];
    std.debug.assert(blk.tier == .ram); // Only demote from RAM

    if (self.ssd_file == null) return error.SsdNotConfigured;

    // Write keys + values to SSD
    const offset = block_id * self.block_bytes;
    try self.ssd_file.?.seekTo(offset);

    const keys_bytes = std.mem.sliceAsBytes(blk.base.keys);
    const values_bytes = std.mem.sliceAsBytes(blk.base.values);
    try self.ssd_file.?.writeAll(keys_bytes);
    try self.ssd_file.?.writeAll(values_bytes);

    // Free RAM backing
    self.allocator.free(blk.base.keys);
    self.allocator.free(blk.base.values);
    blk.base.keys = &[_]f32{};
    blk.base.values = &[_]f32{};

    blk.tier = .ssd;
    blk.ssd_offset = offset;

    std.log.debug("Demoted block {d} from RAM to SSD (offset: {d})", .{block_id, offset});
}
```

/// Restore SSD block to RAM tier.
/// Reads block data from SSD file at stored offset.
pub fn promoteFromSsd(self: *TieredKvCache, block_id: u32) !void {
    var blk = &self.blocks[block_id];
    std.debug.assert(blk.tier == .ssd);

    if (self.ssd_file == null) return error.SsdNotConfigured;

    // Allocate RAM backing
    blk.base.keys = try self.allocator.alloc(f32, self.block_size * self.kv_dim);
    errdefer self.allocator.free(blk.base.keys);
    blk.base.values = try self.allocator.alloc(f32, self.block_size * self.kv_dim);
    errdefer self.allocator.free(blk.base.values);

    // Read from SSD
    const offset = blk.ssd_offset.?;
    try self.ssd_file.?.seekTo(offset);

    const keys_bytes = std.mem.sliceAsBytes(blk.base.keys);
    const values_bytes = std.mem.sliceAsBytes(blk.base.values);
    _ = try self.ssd_file.?.readAll(keys_bytes);
    _ = try self.ssd_file.?.readAll(values_bytes);

    blk.tier = .ram;
    blk.last_access_ms = std.time.milliTimestamp();

    std.log.debug("Promoted block {d} from SSD to RAM", .{block_id});
}
```

**Prefetch job structure:**
```zig
const std = @import("std");
const TieredKvCache = @import("tiered.zig").TieredKvCache;

const PrefetchJob = struct {
    block_id: u32,
};

pub const Prefetcher = struct {
    cache: *TieredKvCache,
    work_queue: std.ArrayList(PrefetchJob),
    mutex: std.Thread.Mutex,
    generation: std.atomic.Value(u64), // For futex wake
    shutdown: std.atomic.Value(bool),
    thread: ?std.Thread = null,
    allocator: std.mem.Allocator,

    pub fn init(allocator: std.mem.Allocator, cache: *TieredKvCache) !Prefetcher {
        return .{
            .cache = cache,
            .work_queue = std.ArrayList(PrefetchJob).init(allocator),
            .mutex = .{},
            .generation = std.atomic.Value(u64).init(0),
            .shutdown = std.atomic.Value(bool).init(false),
            .allocator = allocator,
        };
    }

    pub fn start(self: *Prefetcher) !void {
        self.thread = try std.Thread.spawn(.{}, workerLoop, .{self});
    }

    pub fn deinit(self: *Prefetcher) void {
        self.shutdown.store(true, .release);
        _ = self.generation.fetchAdd(1, .release);
        std.Thread.Futex.wake(&self.generation, 1);

        if (self.thread) |t| t.join();
        self.work_queue.deinit();
    }

    /// Queue prefetch for next N blocks starting from current index.
    /// Called by scheduler during attention compute (per D-07: next 2 blocks).
    pub fn prefetchNext(self: *Prefetcher, block_ids: []const u32, current_idx: usize) !void {
        const prefetch_count: usize = 2; // Per D-07
        const start = current_idx + 1;
        const end = @min(start + prefetch_count, block_ids.len);

        if (start >= block_ids.len) return; // Already at end

        self.mutex.lock();
        defer self.mutex.unlock();

        for (block_ids[start..end]) |block_id| {
            // Only queue if block is in lower tier (SSD or RAM → VRAM)
            if (self.cache.needsPromotion(block_id)) {
                try self.work_queue.append(.{ .block_id = block_id });
            }
        }

        // Wake worker if work added
        if (end > start) {
            _ = self.generation.fetchAdd(1, .release);
            std.Thread.Futex.wake(&self.generation, 1);
        }
    }

    fn workerLoop(self: *Prefetcher) void {
        var local_gen: u64 = 0; // Critical: init to 0, not generation.load

        while (!self.shutdown.load(.acquire)) {
            self.mutex.lock();
            const has_work = self.work_queue.items.len > 0;
            const job = if (has_work) self.work_queue.orderedRemove(0) else null;
            self.mutex.unlock();

            if (job) |j| {
                // Promote block asynchronously
                self.cache.promoteFromSsd(j.block_id) catch |err| {
                    std.log.warn("Prefetch failed for block {d}: {}", .{j.block_id, err});
                };
            } else {
                // No work — sleep on futex
                const current_gen = self.generation.load(.acquire);
                if (current_gen == local_gen) {
                    std.Thread.Futex.wait(&self.generation, current_gen);
                }
                local_gen = self.generation.load(.acquire);
            }
        }
    }
};
```
