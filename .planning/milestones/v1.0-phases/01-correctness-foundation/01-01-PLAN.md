Replace existing Metal SDPA kernel with FlashAttention-2 tiled implementation following the algorithm from RESEARCH.md.

**Block sizes (start conservative, tune later):**
- Bc = 32 (K,V block size)
- Br = 32 (Q,O block size)
- max_d = 256 (maximum head dimension — Gemma3 uses 256, Qwen uses 128)

**Algorithm structure:**
1. One threadgroup per query head (h)
2. Load Q[h*hd : (h+1)*hd] into threadgroup memory q_local[max_d]
3. Initialize online softmax state: m_i = -INFINITY, l_i = 0.0, out_acc[max_d] = 0.0 (register accumulator)
4. Outer loop: iterate over K,V blocks (num_blocks = ceil(sl / Bc))
   a. Load K_block[block_start:block_start+Bc, kvh*hd:(kvh+1)*hd] into threadgroup k_block
   b. Compute scores = Q @ K_block^T (dot product), apply scale
   c. Online softmax max reduction: m_new = threadgroup_reduce_max(scores), update m_i
   d. Rescale previous sum and accumulator: l_i *= exp(m_prev - m_new), out_acc *= exp(m_prev - m_new)
   e. Load V_block into threadgroup v_block
   f. Accumulate: out_acc += exp(scores - m_new) @ V_block, l_i += sum(exp(scores - m_new))
5. Final normalization: output[h*hd + d] = out_acc[d] / l_i
6. Use threadgroup_barrier(mem_flags::mem_threadgroup) after each shared memory write

**GQA head mapping (critical):**
- kvh = h / hpg where hpg = nh / nkv
- K_cache base offset: (block_start + t) * kvd + kvh * hd
- V_cache base offset: (block_start + t) * kvd + kvh * hd

**Threadgroup memory layout (verify fits in 32KB):**
```metal
threadgroup float q_local[max_d];          // 256*4 = 1KB
threadgroup float k_block[32 * 256];       // Bc * max_d = 32KB (full!)
threadgroup float v_block[32 * 256];       // Another 32KB
threadgroup float scores[32 * 32];         // Br * Bc = 4KB
threadgroup float shared[8];               // Reduction scratch = 32 bytes
```
**Memory constraint:** k_block + v_block alone = 64KB, exceeds Metal's ~32KB threadgroup limit. **Solution:** Process K and V sequentially (load K, compute scores, discard K, load V, accumulate, discard V) — never hold both simultaneously. Update algorithm accordingly.

**Kernel signature:**
```metal
kernel void sdpa_fa2(
    device const float* Q,
    device const float* K_cache,
    device const float* V_cache,
    device float* output,
    constant uint& nh,
    constant uint& nkv,
    constant uint& hd,
    constant uint& sl,
    constant float& scale,
    uint h [[threadgroup_position_in_grid]],
    uint tid [[thread_index_in_threadgroup]],
    uint tg_sz [[threads_per_threadgroup]])
```

**Do NOT:**
- Use blit encoders (compute-only per D-02)
- Hardcode head_dim (use `hd` parameter to support both 128 and 256)
- Skip threadgroup barriers (causes race conditions)
- Use naive softmax (use online softmax with running max+sum per D-03)

   ```zig
   pub fn sdpa(
       self: *MetalBackend,
       q: [*]const f32,
       keys: []u8,
       values: []u8,
       k_new: [*]const f32,
       v_new: [*]const f32,
       output: [*]f32,
       nh: usize,
       nkv: usize,
       hd: usize,
       seq_len: usize,
       scale: f32,
       kv_type: KvType,
   ) void {
       // GPU SDPA only supports f32 KV for now
       if (kv_type != .f32) {
           self.cpuFallback().sdpa(q, keys, values, k_new, v_new, output, nh, nkv, hd, seq_len, scale, kv_type);
           return;
       }

       // Get page-aligned buffers for GPU access
       const q_buf = self.getBufRef(@ptrCast(q), nh * hd * 4);
       const k_buf = self.getBufRef(@ptrCast(keys.ptr), keys.len);
       const v_buf = self.getBufRef(@ptrCast(values.ptr), values.len);
       const out_buf = self.getBufRef(@ptrCast(output), nh * hd * 4);

       // Create compute encoder
       const encoder = self.cmd_buf.?.computeCommandEncoder();
       defer encoder.endEncoding();

       encoder.setComputePipelineState(self.sdpa_pipeline.?);
       encoder.setBuffer(q_buf.buf, q_buf.offset, 0);
       encoder.setBuffer(k_buf.buf, k_buf.offset, 1);
       encoder.setBuffer(v_buf.buf, v_buf.offset, 2);
       encoder.setBuffer(out_buf.buf, out_buf.offset, 3);
       encoder.setBytes(&nh, @sizeOf(u32), 4);
       encoder.setBytes(&nkv, @sizeOf(u32), 5);
       encoder.setBytes(&hd, @sizeOf(u32), 6);
       encoder.setBytes(&seq_len, @sizeOf(u32), 7);
       encoder.setBytes(&scale, @sizeOf(f32), 8);

       // Dispatch: nh threadgroups (one per query head), 256 threads per threadgroup
       const threadgroup_size = MTLSize{ .width = 256, .height = 1, .depth = 1 };
       const grid_size = MTLSize{ .width = nh, .height = 1, .depth = 1 };
       encoder.dispatchThreadgroups(grid_size, threadsPerThreadgroup: threadgroup_size);
   }
   ```

4. **Sync handling:**

   - No explicit flush needed here (caller will call be.sync() before reading output)

```zig
const std = @import("std");
const Backend = @import("../src/backend/backend.zig").Backend;
const CpuBackend = @import("../src/backend/cpu.zig").CpuBackend;
const MetalBackend = @import("../src/backend/metal.zig").MetalBackend;

test "Metal SDPA dual-delta correctness" {
    const allocator = std.testing.allocator;

    // Skip if not on macOS
    if (comptime @import("builtin").os.tag != .macos) return error.SkipZigTest;

    // Test parameters (small for fast test, representative for correctness)
    const nh = 4;
    const nkv = 1;  // GQA
    const hd = 128;
    const seq_len = 64;
    const scale = 1.0 / @sqrt(@as(f32, @floatFromInt(hd)));

    // Allocate test data
    var q = try allocator.alloc(f32, nh * hd);
    defer allocator.free(q);
    var keys = try allocator.alloc(f32, seq_len * nkv * hd);
    defer allocator.free(keys);
    var values = try allocator.alloc(f32, seq_len * nkv * hd);
    defer allocator.free(values);
    var cpu_output = try allocator.alloc(f32, nh * hd);
    defer allocator.free(cpu_output);
    var gpu_output = try allocator.alloc(f32, nh * hd);
    defer allocator.free(gpu_output);
    var oracle_output = try allocator.alloc(f64, nh * hd);
    defer allocator.free(oracle_output);

    // Initialize with deterministic random values
    var prng = std.rand.DefaultPrng.init(42);
    const random = prng.random();
    for (q) |*v| v.* = random.float(f32) * 2.0 - 1.0;
    for (keys) |*v| v.* = random.float(f32) * 2.0 - 1.0;
    for (values) |*v| v.* = random.float(f32) * 2.0 - 1.0;

    // Compute FP64 oracle (high-precision CPU reference)
    computeOracleSdpa(q, keys, values, oracle_output, nh, nkv, hd, seq_len, scale);

    // Compute CPU SDPA (f32)
    var cpu_be = try CpuBackend.init(allocator);
    defer cpu_be.deinit();
    var cpu_backend: Backend = .{ .cpu = cpu_be };
    cpu_backend.sdpa(
        q.ptr,
        @ptrCast(keys),
        @ptrCast(values),
        q.ptr + (nh - 1) * hd,  // k_new = last Q head (dummy)
        q.ptr + (nh - 1) * hd,  // v_new = last Q head (dummy)
        cpu_output.ptr,
        nh,
        nkv,
        hd,
        seq_len,
        scale,
        .f32,
    );

    // Compute Metal SDPA (f32 GPU)
    var metal_be = try MetalBackend.init(allocator);
    defer metal_be.deinit();
    var metal_backend: Backend = .{ .metal = metal_be };
    metal_backend.sdpa(
        q.ptr,
        @ptrCast(keys),
        @ptrCast(values),
        q.ptr + (nh - 1) * hd,
        q.ptr + (nh - 1) * hd,
        gpu_output.ptr,
        nh,
        nkv,
        hd,
        seq_len,
        scale,
        .f32,
    );
    metal_backend.sync();  // Download GPU results

    // Dual-delta validation
    var max_cpu_err: f32 = 0.0;
    var max_gpu_err: f32 = 0.0;
    for (0..nh * hd) |i| {
        const cpu_err = @abs(cpu_output[i] - @as(f32, @floatCast(oracle_output[i])));
        const gpu_err = @abs(gpu_output[i] - @as(f32, @floatCast(oracle_output[i])));
        max_cpu_err = @max(max_cpu_err, cpu_err);
        max_gpu_err = @max(max_gpu_err, gpu_err);
    }

    std.debug.print("Metal SDPA dual-delta test:\n", .{});
    std.debug.print("  CPU max error: {e:.2}\n", .{max_cpu_err});
    std.debug.print("  GPU max error: {e:.2}\n", .{max_gpu_err});
    std.debug.print("  GPU/CPU ratio: {d:.2}×\n", .{max_gpu_err / max_cpu_err});

    // Accept GPU if error <= 2× CPU error
    try std.testing.expect(max_gpu_err <= 2.0 * max_cpu_err);
}

fn computeOracleSdpa(
    q: []const f32,
    keys: []const f32,
    values: []const f32,
    output: []f64,
    nh: usize,
    nkv: usize,
    hd: usize,
    seq_len: usize,
    scale: f32,
) void {
    // High-precision FP64 SDPA implementation
    const hpg = nh / nkv;
    for (0..nh) |h| {
        const kvh = h / hpg;
        for (0..hd) |d| {
            var sum: f64 = 0.0;
            var max_score: f64 = -std.math.inf(f64);

            // Compute scores and find max
            var scores: [256]f64 = undefined;  // Max seq_len for test
            for (0..seq_len) |t| {
                var dot: f64 = 0.0;
                for (0..hd) |k| {
                    const q_val: f64 = @floatCast(q[h * hd + k]);
                    const k_val: f64 = @floatCast(keys[t * nkv * hd + kvh * hd + k]);
                    dot += q_val * k_val;
                }
                scores[t] = dot * @as(f64, @floatCast(scale));
                max_score = @max(max_score, scores[t]);
            }

            // Softmax numerator and sum
            var exp_sum: f64 = 0.0;
            for (0..seq_len) |t| {
                scores[t] = @exp(scores[t] - max_score);
                exp_sum += scores[t];
            }

            // Weighted sum over values
            for (0..seq_len) |t| {
                const weight = scores[t] / exp_sum;
                const v_val: f64 = @floatCast(values[t * nkv * hd + kvh * hd + d]);
                sum += weight * v_val;
            }
            output[h * hd + d] = sum;
        }
    }
}
```
